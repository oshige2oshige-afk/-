// --- Hands設定の最適化 --- hands.setOptions({ maxNumHands: 2, // 必ず2本まで探す modelComplexity: 1, // 精度優先（0から1に変更） minDetectionConfidence: 0.4, // 検出感度を少し上げる（0.5→0.4） minTrackingConfidence: 0.4 // 追跡感度を少し上げる（0.5→0.4） }); function onHandResults(res) { handsReady = true; const canvas = document.getElementById('c-hand'); const ctx = canvas.getContext('2d'); if (canvas.width !== 640) { canvas.width = 640; canvas.height = 480; } ctx.clearRect(0, 0, 640, 480); // 毎フレーム、データを一旦リセット state.lx = state.ly = state.rx = state.ry = null; if (res.multiHandLandmarks && res.multiHandLandmarks.length > 0) { // 見つかった手の数だけループ処理 for (let i = 0; i < res.multiHandLandmarks.length; i++) { const lms = res.multiHandLandmarks[i]; const label = res.multiHandedness[i].label; // "Left" or "Right" const x = (lms[9].x * 100).toFixed(1); const y = ((1 - lms[9].y) * 100).toFixed(1); // 左右判定に基づき代入 // ※カメラが鏡像（反転）の場合は label の解釈が逆になることがあります if (label === "Left") { state.lx = x; state.ly = y; } else { state.rx = x; state.ry = y; } // 描画 window.drawConnectors(ctx, lms, window.HAND_CONNECTIONS, { color: label === "Left" ? '#00FF00' : '#FF0000', // 左手を緑、右手を赤で区別 lineWidth: 4 }); window.drawLandmarks(ctx, lms, {color: '#FFFFFF', lineWidth: 1, radius: 2}); } } // UI上の表示更新 const handCount = res.multiHandLandmarks ? res.multiHandLandmarks.length : 0; document.getElementById('hand-tag').innerText = `Hands: ${handCount}`; checkAllReady(); }
